IF $SYS_MOVIE == 1 JMP FULLSIZE
:FULLSIZE
IF $PMOVIE == 0 JMP MOVIE1
IF $PMOVIE == 1 JMP MOVIE2
:MOVIE1
WOPEN 0 MOVIE 0 0 640 480
MVP ccop_an
MVS
WCLOSE 0
WOPEN 0 NOVEL 0 0 640 480
$PMOVIE = 1
JMP MOV_END
:MOVIE2
WOPEN 0 MOVIE 0 0 640 480
MVP ccop_an 1
MVS
WCLOSE 0
WOPEN 0 NOVEL 0 0 640 480
$PMOVIE = 1
JMP MOV_END
:MOV_END
